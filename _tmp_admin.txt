import React, { useEffect, useMemo, useState } from 'react';
import axios from 'axios';

const API_URL = import.meta.env.VITE_API_URL;

const generateLocalPassword = () => {
    const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
    return Array.from({ length: 6 }, () => chars[Math.floor(Math.random() * chars.length)]).join('');
};

export default function AdminPanel({ onBack }) {
    const [teachers, setTeachers] = useState([]);
    const [students, setStudents] = useState([]);
    const [selectedUser, setSelectedUser] = useState(null);
    const [alert, setAlert] = useState('');
    const [form, setForm] = useState({
        role: 'teacher',
        username: '',   // ëîăčí/ID
        first_name: '', // čě˙
        surname: '',    // ôŕěčëč˙
        group_number: '',
        password: '',
    });

    const token = localStorage.getItem('access');
    const axiosConfig = useMemo(() => ({
        headers: { Authorization: `Bearer ${token}` },
    }), [token]);

    const fetchUsers = async () => {
        try {
            const [teacherRes, studentRes] = await Promise.all([
                axios.get(`${API_URL}/api/admin/users/?role=teacher`, axiosConfig),
                axios.get(`${API_URL}/api/admin/users/?role=student`, axiosConfig),
            ]);
            setTeachers(teacherRes.data || []);
            setStudents(studentRes.data || []);
        } catch (error) {
            console.error('Ulanyjylary almakda nasazlyk:', error.response?.data || error);
            setAlert('Maglumatlary almakda nasazlyk boldy.');
        }
    };

    useEffect(() => { fetchUsers(); }, []);

    const handleInputChange = (key, value) => {
        setForm((prev) => ({ ...prev, [key]: value }));
    };

    const handleGeneratePassword = () => {
        handleInputChange('password', generateLocalPassword());
    };

    const handleCreateUser = async () => {
        const usernameTrim = form.username.trim();
        const firstNameTrim = form.first_name.trim();
        const surnameTrim = form.surname.trim();
        const groupTrim = form.group_number.trim();
        const passwordTrim = form.password.trim();

        if (!usernameTrim || !firstNameTrim || !surnameTrim) {
            setAlert('Lojin (ID), ady we familiyasy yazűëmaly.');
            return;
        }
        if (!passwordTrim || passwordTrim.length < 6) {
            setAlert('Parol hokmany we azyndan 6 nysan bolmaly.');
            return;
        }
        if (form.role === 'student') {
            if (!/^[0-9]{6}$/.test(usernameTrim)) {
                setAlert('Student ID dine 6 sanydan durmaly.');
                return;
            }
            if (!groupTrim) {
                setAlert('Topar belgisi yazylmaly.');
                return;
            }
        }

        const payload = {
            username: usernameTrim,
            first_name: firstNameTrim,
            surname: surnameTrim,
            role: form.role,
            group_number: form.role === 'student' ? groupTrim : '',
        };
        payload.password = passwordTrim;

        try {
            const res = await axios.post(`${API_URL}/api/admin/users/create/`, payload, axiosConfig);
            const created = res.data;
            if (created.role === 'teacher') {
                setTeachers((prev) => [...prev, created]);
            } else {
                setStudents((prev) => [...prev, created]);
            }
            setAlert('Ulanyjy ustunlikli gosuldy.');
            setForm({ role: 'teacher', username: '', first_name: '', surname: '', group_number: '', password: '' });
            setSelectedUser(created);
        } catch (error) {
            console.error('Ulanyjy doretmekde nasazlyk:', error.response?.data || error);
            const data = error.response?.data;
            let message = data?.detail;
            if (!message && data) {
                if (typeof data === 'string') message = data;
                else if (Array.isArray(data)) message = data.join(' ');
                else if (typeof data === 'object') message = Object.values(data).flat().join(' ');
            }
            setAlert(message || 'Doretmekde nasazlyk.');
        }
    };

    const handleDeleteUser = async (user) => {
        if (!user?.id) return;
        if (!window.confirm(`"${user.username} ${user.surname}" pozulsynmy?`)) return;
        try {
            await axios.delete(`${API_URL}/api/admin/users/${user.id}/`, axiosConfig);
            if (user.role === 'teacher') {
                setTeachers((prev) => prev.filter((t) => t.id !== user.id));
            } else {
                setStudents((prev) => prev.filter((s) => s.id !== user.id));
            }
            if (selectedUser?.id === user.id) setSelectedUser(null);
            setAlert('Pozuldy.');
        } catch (error) {
            console.error('Pozmakda nasazlyk:', error.response?.data || error);
            setAlert(error.response?.data?.detail || 'Pozup bolmady.');
        }
    };

    const studentsByGroup = useMemo(() => {
        return students.reduce((acc, st) => {
            const key = st.group_number || 'Topar gorkezilmedi';
            if (!acc[key]) acc[key] = [];
            acc[key].push(st);
            return acc;
        }, {});
    }, [students]);

    const renderUserCard = (user) => (
        <div className="test-card card-two-column-layout user-card" key={`${user.role}-${user.id}`}>
                <div className="column-left" onClick={() => setSelectedUser(user)}>
                <div className="tema-wrap">{user.first_name} {user.surname} ({user.username})</div>
                <div className="mugallym-info">
                    <span>{user.role === 'teacher' ? 'Mugallym' : 'Talyp'}</span>
                    {user.group_number && <span className="group-badge">Topar: {user.group_number}</span>}
                </div>
            </div>
            <div className="column-right">
                <button className="btn btn-blue" onClick={() => setSelectedUser(user)}>Maglumat</button>
                <button className="delete-btn" onClick={() => handleDeleteUser(user)}>???</button>
            </div>
        </div>
    );

    return (
        <div className="app-bg">
            <header className="app-header sticky-header">
                <h1 className="header-title">Admin Paneli</h1>
                <button onClick={onBack} className="logout-btn">Cykys</button>
            </header>

            <main className="main-container">
                {alert && <div className="status-message">{alert}</div>}

                <div className="card admin-card">
                    <h3 className="main-title">Taze ulanyjy doretmek</h3>
                    <div className="button-row">
                        <button className={`btn ${form.role === 'teacher' ? 'btn-green' : 'btn-gray'}`} onClick={() => handleInputChange('role', 'teacher')}>Mugallym</button>
                        <button className={`btn ${form.role === 'student' ? 'btn-green' : 'btn-gray'}`} onClick={() => handleInputChange('role', 'student')}>Talyp</button>
                    </div>
                    {form.role === 'student' && (
                        <input
                            className="input-field"
                            placeholder="Talyp ID (6 san)"
                            value={form.username}
                            onChange={(e) => handleInputChange('username', e.target.value)}
                        />
                    )}
                    {form.role === 'teacher' && (
                        <div className="info-text info-gray" style={{ width: '100%', marginBottom: '0.5rem' }}>
                            Lojin awto dorediler: ady+familiyasy (eger gerek bolsa uytgedip bilersiniz).
                        </div>
                    )}
                    <input
                        className="input-field"
                        placeholder="Ady"
                        value={form.first_name}
                        onChange={(e) => handleInputChange('first_name', e.target.value)}
                    />
                    <input
                        className="input-field"
                        placeholder="Familiyasy"
                        value={form.surname}
                        onChange={(e) => handleInputChange('surname', e.target.value)}
                    />
                    {form.role === 'student' && (
                        <input
                            className="input-field"
                            placeholder="Topar belgisi"
                            value={form.group_number}
                            onChange={(e) => handleInputChange('group_number', e.target.value)}
                        />
                    )}
                    <div className="button-row">
                        <input
                            className="input-field"
                            placeholder="Parol (bos bolsa 6 nysanly kod dorediler)"
                            value={form.password}
                            onChange={(e) => handleInputChange('password', e.target.value)}
                        />
                        <button className="btn btn-gray full-width-btn" onClick={handleGeneratePassword}>Kody generirle</button>
                    </div>
                    <button className="btn btn-green full-width-btn" onClick={handleCreateUser}>Ulanyjyny dored</button>
                </div>

                <div className="admin-section">
                    <h3 className="section-title">Mugallymlar</h3>
                    <div className="card-grid">
                        {teachers.length ? teachers.map(renderUserCard) : <p className="no-tests-message">Mugallym yok.</p>}
                    </div>
                </div>

                <div className="admin-section">
                    <h3 className="section-title">Talyp toparlary</h3>
                    {Object.keys(studentsByGroup).length ? (
                        Object.keys(studentsByGroup).map((group) => (
                            <div key={group} className="test-card group-card">
                                <div className="column-left">
                                    <div className="tema-wrap">Topar: {group}</div>
                                    <div className="count">Talyp sany: {studentsByGroup[group].length}</div>
                                </div>
                                <div className="column-right">
                                    <button className="btn btn-blue" onClick={() => setSelectedUser({ group, students: studentsByGroup[group] })}>Gor</button>
                                </div>
                                <div className="group-students">
                                    {studentsByGroup[group].map((st) => (
                                        <span key={st.id} className="chip" onClick={() => setSelectedUser(st)}>
                                            {st.first_name} {st.surname} ({st.username})
                                        </span>
                                    ))}
                                </div>
                            </div>
                        ))
                    ) : (
                        <p className="no-tests-message">Talyp yazgylary yok.</p>
                    )}
                </div>

                {selectedUser && selectedUser.students && (
                    <div className="card">
                        <button className="back-btn-inner" onClick={() => setSelectedUser(null)}>< Yza</button>
                        <h3 className="main-title">Topar: {selectedUser.group}</h3>
                        <div className="group-students">
                            {selectedUser.students.map((st) => (
                                <span key={st.id} className="chip" onClick={() => setSelectedUser(st)}>
                                    {st.first_name} {st.surname} ({st.username})
                                </span>
                            ))}
                        </div>
                    </div>
                )}

                {selectedUser && !selectedUser.students && (
                    <div className="card">
                        <button className="back-btn-inner" onClick={() => setSelectedUser(null)}>< Yza</button>
                        <h3 className="main-title">{selectedUser.first_name} {selectedUser.surname}</h3>
                        <p className="info-text">Roly: {selectedUser.role === 'teacher' ? 'Mugallym' : 'Talyp'}</p>
                        {selectedUser.group_number && <p className="info-text">Topar: {selectedUser.group_number}</p>}
                        <p className="info-text">ID / Lojin: {selectedUser.username}</p>
                        <p className="info-text">Parol / kod: <strong>{selectedUser.generated_password || form.password || '—'}</strong></p>
                        <button className="delete-btn" onClick={() => handleDeleteUser(selectedUser)}>Poz</button>
                    </div>
                )}
            </main>
        </div>
    );
}
